## Usage Examples

### üöÄ Basic AI-Enhanced WebSocket
```typescript
import { useEnterpriseWebSocket } from './hooks/useWebSocket';

const Dashboard = () => {
  const { connectionStatus, aiAnalysis, sendMessage } = useEnterpriseWebSocket('/ws/dashboard/', {
    enableAI: true,
    onAIAnalysis: (analysis) => {
      console.log('AI Analysis:', analysis);
      if (analysis.learning_impact === 'critical') {
        // Immediate intervention required
        showUrgentNotification(analysis.recommended_action);
      }
    }
  });

  const requestUpdate = () => {
    sendMessage({ 
      type: 'request_dashboard_update',
      priority: 'high'
    });
  };

  return (
    <div>
      <StatusIndicator status={connectionStatus} />
      {aiAnalysis && <AIInsights analysis={aiAnalysis} />}
      <UpdateButton onClick={requestUpdate} />
    </div>
  );
};
```

### ü§ù Real-Time Collaborative Learning
```typescript
import { useCollaborativeLearning } from './hooks/useWebSocket';

const CollaborativeMathSession = () => {
  const {
    collaborationSession,
    participants,
    chatMessages,
    sendChatMessage,
    sendRealTimeEdit,
    requestPeerAssessment
  } = useCollaborativeLearning('advanced_calculus_group_1');

  const sendMessage = (content: string) => {
    sendChatMessage(content);
  };

  const makeEdit = (operation: string, position: number, content: string) => {
    sendRealTimeEdit(operation, position, content);
  };

  const assessPeer = (userId: string) => {
    requestPeerAssessment(userId, {
      criteria: ['technical_accuracy', 'problem_solving', 'collaboration']
    });
  };

  return (
    <div className="collaborative-session">
      <ParticipantsList participants={participants} />
      <RealTimeEditor onEdit={makeEdit} />
      <ChatBox messages={chatMessages} onSend={sendMessage} />
      <PeerAssessment onAssess={assessPeer} />
    </div>
  );
};
```
