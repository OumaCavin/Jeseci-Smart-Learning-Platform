```typescript
  ariaLabel="Loading content"
```

### **Progress Loading**
```typescript
const { startLoading, updateProgress } = useLoading();
const loadingId = startLoading({ type: 'progress', estimatedDuration: 5000 });

// Update progress
updateProgress(loadingId, {
  value: 50,
  estimatedTimeRemaining: 25,
  currentStep: "Processing data",
  totalSteps: 4,
  completedSteps: 2
});
```

### **Educational Loading**
```typescript
const { useEducationalLoading } = useLoading();
const loadingId = useEducationalLoading(10000);

// Custom messages
<LoadingSpinner
  type="educational"
  messages={educationalMessages}
  messageIndex={currentMessageIndex}
/>
```

```typescript
const { useAchievementLoading } = useLoading();
```
